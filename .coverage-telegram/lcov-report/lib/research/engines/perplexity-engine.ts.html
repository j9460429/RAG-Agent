import { createPerplexity } from '@ai-sdk/perplexity'
import { generateText } from 'ai'
import type { ResearchReport } from '../research-synthesizer'
import { parseMarkdownToReport, extractSourcesFromText } from './report-parser'
 
interface PerplexityOptions {
  topic: string
  context?: string
  onProgress?: (stage: string, message: string) => void
}
 
/**
 * 使用 Perplexity Sonar Pro 執行進階研究
 *
 * 一次呼叫完成搜尋 + 分析 + 報告產出。
 * 透過 @ai-sdk/perplexity 官方 provider 呼叫。
 */
export async function runPerplexityResearch({
  topic,
  context,
  onProgress,
}: PerplexityOptions): Promise<ResearchReport> {
  const apiKey = process.env.PERPLEXITY_API_KEY
  if (!apiKey) throw new Error('PERPLEXITY_API_KEY 未設定')
 
  const perplexity = createPerplexity({
    apiKey,
  })
 
  onProgress?.('searching', `Perplexity Sonar Pro 正在搜尋「${topic}」的最新資訊...`)
 
  const contextHint = context
    ? `\n\n使用者提供的參考脈絡：${context.slice(0, 2000)}`
    : ''
 
  const { text, providerMetadata } = await generateText({
    model: perplexity('sonar-pro'),
    system: PERPLEXITY_SYSTEM_PROMPT,
    prompt: `研究主題：「${topic}」${contextHint}`,
    experimental_telemetry: {
      isEnabled: true,
      functionId: 'perplexity-research',
      metadata: { feature: 'perplexity-research', topic },
    },
  })
 
  onProgress?.('analyzing', '正在解析搜尋結果，整理來源與報告結構...')
 
  const citations = extractPerplexityCitations(providerMetadata)
  const report = parseMarkdownToReport(text, citations)
 
  onProgress?.('completed', '進階研究完成')
  return report
}
 
/**
 * 從 Perplexity providerMetadata 提取 citations
 * 加 fallback：若取不到結構化 citations，從文字中提取連結
 */
function extractPerplexityCitations(
  metadata: Record<string, Record<string, unknown>> | undefined
): { title: string; url: string }[] {
  if (!metadata) return []
 
  // Perplexity 透過 OpenAI 相容格式回傳 citations
  const perplexityMeta = metadata['perplexity'] as Record<string, unknown> | undefined
  const citations = perplexityMeta?.['citations'] as string[] | undefined
 
  if (citations && citations.length > 0) {
    return citations.map((url, i) => ({
      title: `來源 ${i + 1}`,
      url,
    }))
  }
 
  return []
}
 
/** 提取 citations 並合併文字中的連結（fallback） */
export function extractAllSources(
  text: string,
  metadata: Record<string, Record<string, unknown>> | undefined
): { title: string; url: string }[] {
  const fromMeta = extractPerplexityCitations(metadata)
  if (fromMeta.length > 0) return fromMeta
 
  // fallback: 從文字中提取 Markdown 連結
  return extractSourcesFromText(text)
}
