import { generateText } from "ai";
import { google } from "@ai-sdk/google";
 
import type { SubQuestion } from "./question-decomposer";
 
export interface ResearchFinding {
  subQuestionId: string;
  question: string;
  summary: string;
  keyFindings: string[];
  sources: { title: string; url: string }[];
}
 
/**
 * 並行研究所有子問題
 *
 * 使用 Google Search grounding tool 進行即時網路搜尋，
 * 為每個子問題生成結構化的研究發現。
 */
export async function researchAllQuestions(
  subQuestions: SubQuestion[],
  onProgress?: (
    completed: number,
    total: number,
    current: string,
    currentId?: string,
    completedIds?: string[],
  ) => void,
): Promise<ResearchFinding[]> {
  const results: ResearchFinding[] = [];
  const completedIds: string[] = [];
  const total = subQuestions.length;
 
  // 並行執行所有子問題的研究（每個子問題獨立呼叫 Google Search）
  const promises = subQuestions.map(async (sq) => {
    // 通知開始搜尋此子問題
    onProgress?.(completedIds.length, total, sq.question, sq.id, [
      ...completedIds,
    ]);
    const finding = await researchSingleQuestion(sq);
    results.push(finding);
    completedIds.push(sq.id);
    onProgress?.(completedIds.length, total, sq.question, undefined, [
      ...completedIds,
    ]);
    return finding;
  });
 
  await Promise.all(promises);
 
  // 按照原始子問題順序排列結果
  const orderedResults = subQuestions.map(
    (sq) => results.find((r) => r.subQuestionId === sq.id)!,
  );
 
  return orderedResults;
}
 
async function researchSingleQuestion(
  sq: SubQuestion,
): Promise<ResearchFinding> {
  try {
    const { text } = await generateText({
      model: google("gemini-3.1-pro-preview"),
      // @ts-expect-error ai sdk v6 maxSteps type defs
      maxSteps: 5,
      tools: {
        googleSearch: google.tools.googleSearch({}),
      },
      experimental_telemetry: {
        isEnabled: true,
        functionId: "parallel-research-question",
        metadata: { feature: "deep-research", question: sq.question },
      },
      prompt: `You are a senior research analyst with expertise in deep, thorough investigation. Research the following question using Google Search and provide a **comprehensive, data-rich** answer with maximum depth.
 
Question: ${sq.question}
Search keywords to use: ${sq.searchQuery}
 
## Search Strategy (MUST follow):
1. **Round 1**: Search with the provided keywords to get baseline information
2. **Round 2**: Search with ALTERNATIVE keyword variations (synonyms, related terms, industry jargon) to find additional perspectives
3. **Round 3**: Search for SPECIFIC DATA — statistics, market size, growth rates, adoption rates, recent reports
4. **Round 4** (if needed): Search for CASE STUDIES and real-world examples
5. Aim to use the googleSearch tool AT LEAST 3-4 TIMES with different queries
 
## Output Requirements:
1. Synthesize findings from AT LEAST 5-8 different high-quality sources
2. EVERY finding must include specific numbers, statistics, dates, or concrete examples
3. Cross-reference information between sources — note agreements and contradictions
4. Prioritize recent data (within the last 1-2 years)
5. Provide your response in the following format (use Traditional Chinese for content):
 
## 摘要
[8-10 句詳細摘要，涵蓋：
- 問題的背景與重要性
- 3-4 個關鍵發現（含具體數據）
- 整體趨勢判斷
- 值得特別關注的亮點或轉折]
 
## 關鍵發現
- [發現 1：具體描述 + 數據/統計 + 來源標注]
- [發現 2：具體描述 + 數據/統計 + 來源標注]
- [發現 3：具體描述 + 數據/統計 + 來源標注]
- [發現 4：具體描述 + 數據/統計 + 來源標注]
- [發現 5：具體描述 + 數據/統計 + 來源標注]
- [發現 6：具體描述 + 數據/統計 + 來源標注]
- [發現 7：如有]
- [發現 8：如有]
 
## 矛盾與爭議
[記錄不同來源間的矛盾觀點或爭議，例如：
- 來源A認為...，但來源B指出...
- 這些分歧的可能原因是...]
 
## 來源
- [Source title 1](URL1)
- [Source title 2](URL2)
- [Source title 3](URL3)
- [Source title 4](URL4)
- [Source title 5](URL5)
- [Source title 6](URL6)
- [Source title 7](URL7)
- [Source title 8](URL8)`,
    });
 
    return parseResearchResponse(sq, text);
  } catch (err) {
    console.error(
      `[Research] Failed to research "${sq.question}":`,
      err instanceof Error ? err.message : err,
    );
    return {
      subQuestionId: sq.id,
      question: sq.question,
      summary: "研究過程中發生錯誤，無法取得相關資訊。",
      keyFindings: [],
      sources: [],
    };
  }
}
 
function parseResearchResponse(sq: SubQuestion, text: string): ResearchFinding {
  // 提取摘要
  const summaryMatch = text.match(/##\s*摘要\s*\n([\s\S]*?)(?=\n##|$)/);
  const summary = summaryMatch?.[1]?.trim() ?? text.slice(0, 300);
 
  // 提取關鍵發現
  const findingsMatch = text.match(/##\s*關鍵發現\s*\n([\s\S]*?)(?=\n##|$)/);
  const findingsBlock = findingsMatch?.[1] ?? "";
  const keyFindings = findingsBlock
    .split("\n")
    .filter((line) => line.trim().startsWith("-"))
    .map((line) => line.replace(/^-\s*/, "").trim())
    .filter(Boolean);
 
  // 提取來源
  const sourcesMatch = text.match(/##\s*來源\s*\n([\s\S]*?)$/);
  const sourcesBlock = sourcesMatch?.[1] ?? "";
  const sources = sourcesBlock
    .split("\n")
    .map((line) => {
      const match = line.match(/\[([^\]]+)\]\(([^)]+)\)/);
      if (match) return { title: match[1], url: match[2] };
      return null;
    })
    .filter((s): s is { title: string; url: string } => s !== null);
 
  return {
    subQuestionId: sq.id,
    question: sq.question,
    summary,
    keyFindings: keyFindings.length > 0 ? keyFindings : [summary],
    sources,
  };
}
