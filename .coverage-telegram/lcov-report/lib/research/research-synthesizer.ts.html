import { generateText } from "ai";
import { google } from "@ai-sdk/google";
import type { DecompositionResult } from "./question-decomposer";
import type { ResearchFinding } from "./parallel-researcher";
 
export interface ResearchReport {
  title: string;
  executiveSummary: string;
  sections: { heading: string; content: string }[];
  conclusion: string;
  sources: { title: string; url: string }[];
  generatedAt: string;
}
 
/**
 * 綜合所有研究發現，生成結構化研究報告
 *
 * 使用 Gemini Pro 將分散的子問題答案整合為
 * 具有執行摘要、分節分析、結論建議的完整報告。
 */
export async function synthesizeReport(
  decomposition: DecompositionResult,
  findings: ResearchFinding[],
): Promise<ResearchReport> {
  const findingsContext = findings
    .map((f) => {
      const sources = f.sources
        .map((s) => `  - [${s.title}](${s.url})`)
        .join("\n");
      const keyFindings = f.keyFindings.map((kf) => `  - ${kf}`).join("\n");
      return `### ${f.question}\n摘要：${f.summary}\n關鍵發現：\n${keyFindings}\n來源：\n${sources}`;
    })
    .join("\n\n");
 
  const { text } = await generateText({
    model: google("gemini-3.1-pro-preview"),
    experimental_telemetry: {
      isEnabled: true,
      functionId: "research-synthesizer",
      metadata: { feature: "deep-research", topic: decomposition.mainTopic },
    },
    prompt: `你是一位頂尖的研究分析顧問，擅長將多個子研究結果融合為一份深度、專業、有獨特洞察力的研究報告。你的核心任務是**跨子問題交叉分析**，發現子研究中未明言的關聯性與深層趨勢。
 
研究主題：${decomposition.mainTopic}
研究範圍：${decomposition.researchScope}
 
各子問題研究結果：
${findingsContext}
 
## 合成策略（在撰寫前先思考）
1. 識別各子研究間的**共通主題與矛盾點**
2. 找出**跨子問題的因果關係**（A 子問題的發現如何影響 B 子問題的結論）
3. 整合為**超越個別子問題**的全局性洞察
4. 確保每個論點有**至少 2 個來源**交叉佐證
 
## 報告格式要求（使用繁體中文，嚴格遵守）
 
# [報告標題 — 具體、有吸引力、反映核心發現]
 
## 執行摘要
[300-500 字的結構化摘要，包含：
- 研究背景與重要性（為何此主題值得深入研究）
- 3-5 個核心發現（每個附帶關鍵數據，例如「市場規模達 XX 億美元，年增 XX%」）
- 最值得關注的趨勢或轉折點
- 最高優先級的 1-2 條建議
此摘要應讓讀者僅讀此節即可掌握報告 80% 的價值]
 
## [第一節 — 產業背景與市場全景分析]
[400-600 字，融合多個子研究的背景資訊：
- 產業/技術發展歷程的關鍵里程碑
- 當前市場規模、成長率、滲透率等量化指標（引用具體數據）
- 主要參與者/競爭格局（領導者、挑戰者、新進者）
- 全球市場 vs 亞太/台灣市場的差異對比
- 驅動成長的核心因素（技術突破、政策推動、需求變化）
- 與前一年度的關鍵變化]
 
## [第二節 — 核心議題深度剖析]
[400-600 字，跨子問題交叉分析：
- 從技術面、商業面、使用者面三個維度深入分析
- 正方觀點 vs 反方觀點的詳細對比（含各自的數據支撐）
- 不同利害關係人（企業、消費者、政府、投資者）的立場與考量
- 與競爭方案/替代技術的結構化比較
- 2-3 個具體案例的深入解析（做法、成效、關鍵因素）]
 
## [第三節 — 數據洞察與趨勢分析]
[400-600 字，量化導向：
- 關鍵指標的歷史趨勢描述（過去 3 年的變化）
- 投資/融資動態（近期重要交易與金額）
- 採用率與市場成熟度評估
- 區域市場差異分析
- 短期（1年）與中期（3-5年）趨勢預測，附推論依據]
 
## [第四節 — 挑戰、風險與機會評估]
[400-600 字，SWOT 視角的系統性分析：
- 主要技術挑戰與瓶頸
- 法規/合規/倫理風險
- 市場風險（競爭加劇、需求轉移、替代威脅）
- 被低估的新興機會點
- 進入門檻/轉換成本分析
- 風險緩解策略框架]
 
## [第五節 — 標竿案例與最佳實踐]
[300-500 字，實踐導向：
- 2-3 個成功案例的詳細分析（背景、做法、成效數據、關鍵成功因素）
- 1 個失敗/教訓案例（原因分析、可汲取的教訓）
- 可複製的最佳實踐模式
- 對研究主題的具體啟示]
 
## [第六節 — 跨領域影響與生態系分析]（如研究內容支持）
[300-500 字，宏觀視角：
- 對上下游產業鏈的影響
- 對就業/人才市場的影響
- 社會/倫理層面的考量
- 與其他趨勢（AI、永續發展等）的交互作用]
 
## 結論與建議
[基於所有研究發現的系統性建議：
1. [最高優先] 建議一 — 理由、預期效益、實施要點
2. [高優先] 建議二 — 理由、預期效益、實施要點
3. [高優先] 建議三 — 理由、預期效益、實施要點
4. [中優先] 建議四 — 理由、預期效益、實施要點
5. [中優先] 建議五 — 理由、預期效益、實施要點
 
**行動時間表建議**：
- 即刻行動（0-3個月）：...
- 短期佈局（3-12個月）：...
- 中長期策略（1-3年）：...]
 
## 參考來源
- [來源標題](URL)
- ...
 
## 撰寫品質要求（嚴格遵守）
1. **深度優先**：每個章節至少 400-600 字，禁止表面化描述
2. **數據密集**：每個論點至少 2 個數據點支撐，每 100 字至少引用 1 次來源
3. **融合分析**：禁止簡單羅列子問題答案，必須跨子問題交叉分析、找出關聯
4. **引用規範**：引用來源時使用 [來源標題](URL) 格式
5. **結論嚴謹**：結論必須可追溯至報告中的具體發現，不得臆測
6. **格式清晰**：善用小標題、列點、粗體、比較表格提升可讀性
7. **觀點平衡**：呈現多元觀點，標注各觀點的支持者與依據
8. **獨特洞察**：報告應包含至少 2-3 個「超越常識」的深層洞察`,
  });
 
  return parseReport(text, findings);
}
 
function parseReport(
  text: string,
  findings: ResearchFinding[],
): ResearchReport {
  // 提取標題
  const titleMatch = text.match(/^#\s+(.+)/m);
  const title = titleMatch?.[1]?.trim() ?? "深度研究報告";
 
  // 提取執行摘要（支援帶編號格式如 "## 1. 執行摘要"）
  const summaryMatch = text.match(
    /##\s*(?:\d+[\.\、]\s*)?執行摘要\s*\n([\s\S]*?)(?=\n##|$)/,
  );
  const executiveSummary = summaryMatch?.[1]?.trim() ?? "";
 
  // 提取各節內容（排除執行摘要、結論、參考來源、撰寫品質要求等 meta 區塊）
  const sectionRegex =
    /##\s+(?!(?:\d+[\.\、]\s*)?(?:執行摘要|結論|參考來源|撰寫品質要求))(.+)\n([\s\S]*?)(?=\n##|$)/g;
  const sections: { heading: string; content: string }[] = [];
  let match;
  const excludeKeywords = ["參考", "結論", "執行摘要", "撰寫品質", "合成策略"];
  while ((match = sectionRegex.exec(text)) !== null) {
    const heading = match[1].trim();
    const content = match[2].trim();
    if (
      heading &&
      content &&
      !excludeKeywords.some((kw) => heading.includes(kw))
    ) {
      sections.push({ heading, content });
    }
  }
 
  // 提取結論（支援帶編號格式）
  const conclusionMatch = text.match(
    /##\s*(?:\d+[\.\、]\s*)?結論[與和]?建議?\s*\n([\s\S]*?)(?=\n##|$)/,
  );
  const conclusion = conclusionMatch?.[1]?.trim() ?? "";
 
  // 匯總所有來源（去重）
  const allSources = findings.flatMap((f) => f.sources);
  const uniqueSources = new Map<string, { title: string; url: string }>();
  for (const source of allSources) {
    if (!uniqueSources.has(source.url)) {
      uniqueSources.set(source.url, source);
    }
  }
 
  return {
    title,
    executiveSummary,
    sections,
    conclusion,
    sources: [...uniqueSources.values()],
    generatedAt: new Date().toISOString(),
  };
}
