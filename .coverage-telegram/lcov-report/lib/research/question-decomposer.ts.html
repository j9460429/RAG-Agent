import { generateObject, streamObject } from "ai";
import { z } from "zod";
import { google } from "@ai-sdk/google";
 
const SubQuestionSchema = z.object({
  id: z.string().describe("子問題 ID，格式如 Q1, Q2"),
  question: z.string().describe("具體的研究子問題"),
  searchQuery: z.string().describe("用於 Google Search 的英文搜尋關鍵字"),
  priority: z.enum(["high", "medium", "low"]).describe("優先級"),
});
 
const DecompositionSchema = z.object({
  mainTopic: z.string().describe("研究主題的一句話摘要"),
  subQuestions: z
    .array(SubQuestionSchema)
    .min(5)
    .max(8)
    .describe("拆解後的 MECE 子問題（5 到 8 個）"),
  researchScope: z.string().describe("研究範圍說明，包含限制條件"),
});
 
export type SubQuestion = z.infer<typeof SubQuestionSchema>;
export type DecompositionResult = z.infer<typeof DecompositionSchema>;
 
/** streamObject 串流中的 partial decomposition（所有欄位都可能 undefined） */
export type PartialDecomposition = Partial<{
  mainTopic: string;
  subQuestions: Partial<SubQuestion>[];
  researchScope: string;
}>;
 
const DECOMPOSITION_PROMPT = (
  topic: string,
  contextHint: string,
) => `你是一位專業的研究分析師，擅長將複雜主題系統性拆解為可獨立研究的子問題。請將以下研究主題拆解為 5 到 8 個深度子問題（根據主題複雜度決定數量）。
 
## 分解原則
1. **MECE 原則**：子問題互不重疊（Mutually Exclusive）、完全覆蓋（Collectively Exhaustive）
2. **多維度覆蓋**：子問題應涵蓋以下維度中最相關的 5-8 個：
   - 產業/市場現況與規模（市場有多大？成長多快？）
   - 核心技術/方法深度分析（它怎麼運作？有什麼突破？）
   - 競爭格局與主要參與者（誰在做？做得如何？）
   - 挑戰、風險與法規考量（有什麼障礙？）
   - 趨勢預測與未來機會（接下來會怎樣？）
   - 實務案例與最佳實踐（有誰成功了？怎麼做到的？）
   - 使用者/消費者觀點與需求分析（用戶怎麼想？）
   - 商業模式與獲利策略（怎麼賺錢？可持續嗎？）
3. **可搜尋性**：每個子問題都必須能透過 Google Search 獨立找到豐富資料
 
## 搜尋關鍵字策略
- searchQuery 必須是英文，針對 Google 搜尋優化
- 每個 searchQuery 應包含 3-5 個關鍵字
- 加入時間限定詞（2024, 2025, latest, recent）以獲取最新資料
- 加入量化限定詞（market size, growth rate, statistics, report）以獲取數據
- 使用專業術語變體（例如同時考慮 AI / artificial intelligence / machine learning）
 
## 優先級分配
- **high**（2-3 個）：核心問題，直接回答「這個主題最重要的是什麼」
- **medium**（2-3 個）：補充分析，提供完整全貌所需的脈絡
- **low**（1-2 個）：背景/趨勢，提供前瞻性視角
 
## 輸出要求
- subQuestions 和 mainTopic、researchScope 使用繁體中文
- mainTopic 應精確概括研究主題
- researchScope 應描述研究的邊界與聚焦範圍（80-150 字）
 
研究主題：「${topic}」${contextHint}`;
 
/**
 * 將研究主題拆解為 MECE 子問題（串流版本）
 *
 * 用 Gemini Pro 進行結構化拆解，透過 streamObject 即時串流 partial results。
 * 每當有新的子問題產出時呼叫 onPartial callback，讓前端即時顯示。
 */
export async function decomposeResearchTopic(
  topic: string,
  context?: string,
  onPartial?: (partial: PartialDecomposition) => void,
): Promise<DecompositionResult> {
  const contextHint = context
    ? `\n\n使用者額外脈絡：${context.slice(0, 2000)}`
    : "";
 
  const result = streamObject({
    model: google("gemini-3.1-pro-preview"),
    schema: DecompositionSchema,
    prompt: DECOMPOSITION_PROMPT(topic, contextHint),
  });
 
  // 串流 partial objects，讓呼叫端能即時推送到前端
  if (onPartial) {
    let lastSubQuestionCount = 0;
    for await (const partial of result.partialObjectStream) {
      const currentCount = partial.subQuestions?.length ?? 0;
      // 僅在 mainTopic 首次出現或子問題數量增加時才通知（避免過度推送）
      if (
        (partial.mainTopic &&
          lastSubQuestionCount === 0 &&
          currentCount === 0) ||
        currentCount > lastSubQuestionCount
      ) {
        onPartial(partial as PartialDecomposition);
        lastSubQuestionCount = currentCount;
      }
    }
  }
 
  // 取得最終完整且通過 schema 驗證的結果
  const finalObject = await result.object;
  return finalObject;
}
