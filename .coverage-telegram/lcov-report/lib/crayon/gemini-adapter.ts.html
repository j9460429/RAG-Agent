import { TextChunk, ResponseTemplate } from "@crayonai/stream";
import { parse as bestEffortParse } from "best-effort-json-parser";
import type { TextStreamPart } from "ai";
 
/**
 * Convert a Vercel AI SDK streamText().fullStream (AsyncIterable<TextStreamPart>)
 * to a Crayon-compatible SSE ReadableStream.
 *
 * Modified to handle tool calls (specifically googleSearch) to emit a thinking process template.
 */
 
interface ResponsePart {
  type: string;
  text?: string;
  name?: string;
  templateProps?: Record<string, unknown>;
}
 
/**
 * 多層 JSON 結構偵測 — 判斷 rawAccumulated 中是否已出現 JSON 結構的跡象。
 * ⚠️ 此函式是 RAG 安全的關鍵防線，修改需極度謹慎。
 */
function isJsonStructureDetected(
  rawAccumulated: string,
  stripped: string,
): boolean {
  return (
    /"response"\s*:/.test(rawAccumulated) ||
    stripped.startsWith("{") ||
    /\n\s*{\s*"/.test(rawAccumulated) ||
    /\n\s*\{\s*$/.test(rawAccumulated) || // 末尾的 { 在換行後，高機率是 JSON block 開始，緩衝之
    /```(?:json)?\s*\n?\s*\{/.test(rawAccumulated) ||
    rawAccumulated.includes("```") // 偵測到任何 code fence，暫停以防萬一
  );
}
 
export function stripCodeFences(raw: string): string {
  return raw.replace(/^[\s]*```(?:json)?\s*\n?/, "").replace(/\n?```\s*$/, "");
}
 
export function extractResponseCandidate(raw: string, final = false): string {
  const noFence = stripCodeFences(raw);
  let responseKeyIdx = noFence.search(/"response"\s*:/);
 
  if (responseKeyIdx < 0) {
    // Fallback: Gemini sometimes outputs bare { "type": "template", ... } without the "response" array
    responseKeyIdx = noFence.search(/"type"\s*:\s*"(?:template|text)"/);
    if (responseKeyIdx < 0) return ""; // return empty instead of noFence to prevent "no parser registered"
  }
 
  const objectStart = noFence.lastIndexOf("{", responseKeyIdx);
  if (objectStart < 0) return "";
 
  const candidate = noFence.slice(objectStart);
  if (!final) return candidate;
 
  const lastBrace = candidate.lastIndexOf("}");
  return lastBrace >= 0 ? candidate.slice(0, lastBrace + 1) : candidate;
}
 
/**
 * 解析混合格式回覆：前綴純文字 Markdown + 尾部 JSON `{ "response": [...] }`。
 * 當 Google Search 被觸發時，Gemini 可能不完全遵守 JSON 格式約束，
 * 回覆的前半段是純 Markdown，後半段才是 JSON 結構。
 * 此函式將前綴文字包裝成 text part，並與 JSON 中的 parts 合併。
 */
export function parseHybridResponse(raw: string): ResponsePart[] | null {
  const noFence = stripCodeFences(raw);
  let responseKeyIdx = noFence.search(/"response"\s*:/);
  let isBareObject = false;
 
  if (responseKeyIdx < 0) {
    // Fallback for bare object
    responseKeyIdx = noFence.search(/"type"\s*:\s*"(?:template|text)"/);
    if (responseKeyIdx >= 0) {
      isBareObject = true;
    } else {
      return null;
    }
  }
 
  const objectStart = noFence.lastIndexOf("{", responseKeyIdx);
  if (objectStart < 0) return null;
 
  // 提取 JSON 前的純文字前綴
  let prefix = noFence.slice(0, objectStart).trim();
 
  // 修復：移除前綴末尾的 code fence opening (e.g. ```json)
  // 這是為了防止混合模式下，JSON 被包在 code block 中時，code fence 洩漏到前綴文字中
  prefix = prefix.replace(/```(?:json)?\s*$/, "").trimEnd();
 
  // 提取 JSON 部分
  const jsonCandidate = noFence.slice(objectStart);
  const lastBrace = jsonCandidate.lastIndexOf("}");
  const jsonStr =
    lastBrace >= 0 ? jsonCandidate.slice(0, lastBrace + 1) : jsonCandidate;
 
  // ⚠️ best-effort-json-parser 對 "response":。 等非 JSON token 會拋出 SyntaxError（非返回 null）
  // 必須 try/catch，否則 throw 會傳播到 fromGeminiStream → controller.error() → ERR_EMPTY_RESPONSE
  let parsed: ReturnType<typeof bestEffortParse> | null = null;
  try {
    parsed = bestEffortParse(jsonStr);
  } catch {
    // 嘗試 sanitize 後再解析一次（處理 "key":, 等損壞格式）
    const sanitizedJson = sanitizeMalformedJson(jsonStr);
    if (sanitizedJson !== jsonStr) {
      try {
        parsed = bestEffortParse(sanitizedJson);
      } catch {
        // sanitize 後仍無法解析，返回 null
      }
    }
  }
 
  let jsonParts: ResponsePart[] = [];
 
  if (isBareObject) {
    // If it was a bare object, `parsed` itself is the ResponsePart (or close to it)
    // We expect it to have a `type` property.
    if (parsed && typeof parsed.type === 'string') {
      jsonParts = [parsed as ResponsePart];
    } else {
      return null;
    }
  } else {
    if (!Array.isArray(parsed?.response)) return null;
    jsonParts = parsed!.response as ResponsePart[];
  }
 
  // 如果沒有前綴文字，直接返回 JSON parts
  if (!prefix) return jsonParts;
 
  // 有前綴文字：包裝成 text part 插入最前面
  return [{ type: "text", text: prefix }, ...jsonParts];
}
 
/**
 * 修復 Gemini 偶發的損壞 JSON 格式：
 * - `"key":,`  → `"key":null,`   （value 缺失，逗號接續）
 * - `"key":}`  → `"key":null}`   （value 缺失，物件閉合）
 * 這些模式在 Gemini 串流截斷或 grounding metadata 混入時偶發出現。
 *
 * ⚠️ 注意：不補全 `"key":\n` 的情況 —— JSON 允許在冒號後換行再填值（如陣列）。
 *    舊版將 \n 也視為缺值，導致 `"response":\n[...]` 被錯誤轉成 `"response":null\n[...]`，
 *    造成 Gemini Pro 的 response 欄位被吞掉（UI 顯示「本次回覆未成功生成內容」）。
 *
 * ⚠️ Regex 說明：使用 `("key"\s*:\s*)` 作為 group 1（包含完整的 key + 冒號）
 * 而非 `("key")` — 後者會因 regex 回溯導致 key 字元被截斷（如 "headers" → ""）
 */
function sanitizeMalformedJson(raw: string): string {
  // 只補全緊接 , 或 } 或 ] 的缺值（確定缺值的情況），不補全換行後的值（合法 JSON 多行格式）
  return raw.replace(/("[^"]+"\s*:\s*)([,}\]])/g, "$1null$2");
}
 
export function parseResponseParts(raw: string): ResponsePart[] | null {
  // 1. 嘗試完整 JSON 解析
  // ⚠️ best-effort-json-parser 對 "key":, 格式會拋出 SyntaxError（非返回 null）
  // 必須 try/catch，否則 throw 會傳播到 fromGeminiStream → controller.error() → ERR_EMPTY_RESPONSE
  //
  // 效能優化：只有 { 或 [ 開頭的字串才有可能是 JSON，其他（如 Markdown 的 **text**）
  // 直接跳過，避免 best-effort-json-parser 打出 "no parser registered for '* *'" 的 console.error
  const trimmedRaw = raw.trimStart();
  let direct: ReturnType<typeof bestEffortParse> | null = null;
  if (trimmedRaw.startsWith("{") || trimmedRaw.startsWith("[")) {
    try {
      direct = bestEffortParse(raw);
    } catch {
      // 解析失敗（如損壞的 JSON），繼續嘗試 sanitize path
    }
  }
  if (Array.isArray(direct?.response))
    return direct!.response as ResponsePart[];
 
  // 1b. 若解析失敗或拋出，嘗試修復損壞格式後再解析（同樣只適用於 JSON 開頭字串）
  if (trimmedRaw.startsWith("{") || trimmedRaw.startsWith("[")) {
    const sanitized = sanitizeMalformedJson(raw);
    if (sanitized !== raw) {
      let repaired: ReturnType<typeof bestEffortParse> | null = null;
      try {
        repaired = bestEffortParse(sanitized);
      } catch {
        // sanitize 後仍無法解析
      }
      if (Array.isArray(repaired?.response))
        return repaired!.response as ResponsePart[];
    }
  }
 
  // 2. 嘗試混合格式解析（前綴 Markdown + 尾部 JSON）
  // 優先於 extractResponseCandidate，因為後者會丟棄前綴
  const hybrid = parseHybridResponse(raw);
  if (hybrid) return hybrid;
 
  // 3. 嘗試提取 JSON candidate（處理 code fence 等）
  const extracted = extractResponseCandidate(raw, true);
  const sanitizedExtracted = sanitizeMalformedJson(extracted);
  let parsed: ReturnType<typeof bestEffortParse> | null = null;
  try {
    parsed = bestEffortParse(sanitizedExtracted);
  } catch {
    // 仍無法解析
  }
  if (Array.isArray(parsed?.response))
    return parsed!.response as ResponsePart[];
 
  return null;
}
 
export interface RAGMetadataForStream {
  originalQuery: string;
  finalQuery: string;
  rewrites: number;
  relevanceScore: number;
  relevanceVerdict: string;
  retrievalMethod: "local" | "web" | "hybrid";
  documents: Array<{
    title: string;
    similarity: number;
    sourceType?: "內部" | "外部" | "未知";

  hasGraphContext?: boolean;
}
 
export function fromGeminiStream(
  eventStream: AsyncIterable<TextStreamPart<any>>,
  ragMetadata?: RAGMetadataForStream | null,
): ReadableStream {
  const state: { controller: ReadableStreamDefaultController<string> | null } =
    { controller: null };
 
  const readableStream = new ReadableStream<string>({
    start(c) {
      state.controller = c;
    },
  });
 
  (async () => {
    try {
      let rawAccumulated = ""; // 原始累積（含 code block markers）
      let hasCodeBlock = false;
      let detectionDone = false;
      let sentPartsCount = 0;
      let sentTextLength = 0;
 
      // 直接文字串流：JSON 結構尚未就緒前，逐 chunk 發送原始文字
      let directStreamSentLength = 0;
      let structuredModeActive = false; // JSON 結構已成功解析後切換為結構化模式
 
      // Track if we have already emitted a thinking process to avoid duplicates
      let thinkingEmitted = false;
 
      // P1: JSON 解析節流 — 減少每 token 的 O(n²) 解析開銷
      let charsSinceLastParse = 0;
      const PARSE_THROTTLE_CHARS = 25; // 每累積 N 字元才做一次完整 JSON 解析
 
      const enqueueSSE = (sseString: string) => {
        state.controller?.enqueue(sseString);
      };
 
      // 如有 RAG metadata，在串流開始前先發送
      if (ragMetadata && ragMetadata.documents.length > 0) {
        enqueueSSE(
          new ResponseTemplate(
            "rag_transparency",
            ragMetadata as unknown as Record<string, unknown>,
          ).toSSEString(),
        );
      }
 
      const getCleanContent = (): string => {
        if (!hasCodeBlock) return extractResponseCandidate(rawAccumulated);
        return extractResponseCandidate(rawAccumulated);
      };
 
      // 從原始累積中提取純文字內容（JSON 結構外的 fallback）
      // 關鍵改進：即使 JSON 字串值未閉合（串流截斷），也能提取已收到的文字
      // 額外支援混合格式：前綴 Markdown + 尾部 JSON
      const extractRawTextContent = (): string => {
        const raw = rawAccumulated;
 
        // 找到最後一個 "text": " 開頭的位置，提取到字串尾端或下一個未轉義的引號
        // 支援多個 text part（取所有 "text":"..." 的內容拼接）
        let result = "";
        const keyPattern = /"text"\s*:\s*"/g;
        let match: RegExpExecArray | null;
 
        const processSegment = (seg: string): string => {
          try {
            return JSON.parse(`"${seg}"`);
          } catch {
            return seg
              .replace(/\\n/g, "\n")
              .replace(/\\t/g, "\t")
              .replace(/\\"/g, '"')
              .replace(/\\\\/g, "\\")
              .replace(/\\u([0-9a-fA-F]{4})/g, (_, hex) =>
                String.fromCharCode(parseInt(hex, 16)),
              );
          }
        };
 
        while ((match = keyPattern.exec(raw)) !== null) {
          const valueStart = match.index + match[0].length;
          // 從 valueStart 往後掃描，找到未轉義的 " 或字串尾端
          let i = valueStart;
          let escaped = false;
          while (i < raw.length) {
            const ch = raw[i];
            if (escaped) {
              escaped = false;
              i++;
              continue;
            }
            if (ch === "\\") {
              escaped = true;
              i++;
              continue;
            }
            if (ch === '"') break;
            i++;
          }
          // 判斷是否為串流截斷（未找到閉合引號）
          const isTruncated = i === raw.length;
 
          const segment = raw.slice(valueStart, i);
 
          if (segment.length > 0) {
            // 🔥 Escape Sequence Buffering Fix:
            // 如果 segment 以 \ 結尾，且是因為截斷造成的，則暫緩輸出最後一個 \
            // 等待下一個 chunk 補完（例如 \n, \", \\）
            // 這樣可以避免 \ 和 n 被分開處理導致換行符失效
            if (
              isTruncated &&
              segment.endsWith("\\") &&
              !segment.endsWith("\\\\")
            ) {
              // 檢查是否為偶數個反斜線（即結尾是真的轉義符，而不是轉義的反斜線）
              // 簡單起見，我們只處理最常見的單個 \ 結尾情況
              // 如果是 ...\\，則是完整的反斜線字符，不需要緩衝
              // 如果是 ...\\\，則是反斜線 + 轉義起始，需要緩衝
              let backslashCount = 0;
              for (let j = segment.length - 1; j >= 0; j--) {
                if (segment[j] === "\\") backslashCount++;
                else break;
              }
              if (backslashCount % 2 === 1) {
                // 奇數個反斜線 -> 最後一個是轉義起始 -> 緩衝它（不包含在 result 中）
                const safeSegment = segment.slice(0, -1);
                result += processSegment(safeSegment);
                // 跳過後續處理
                continue;
              }
            }
 
            result += processSegment(segment);
          }
        }
 
        // 混合格式支援...
        // 且原始文字不是以 JSON 開頭（即混合格式的前綴 Markdown），
        // 則返回 JSON 結構開始前的前綴文字作為 fallback
        // ⚠️ 關鍵：不能返回整段 raw（包含 JSON 原始碼），否則 JSON 會洩漏到前端
        if (!result && raw.length > 0) {
          // 🔥 Partial Code Fence Fix:
          // 若 rawAccumulated 以 1-2 個 backtick 開頭（不完整的 code fence），
          // stripCodeFences regex 無法識別（需要三個 backtick），
          // 這些字元會被誤當作純文字 emit 到前端。
          // 解法：若開頭符合 `^`{1,2}(?:json)?\s*$` 的不完整 code fence，暫緩 emit。
          const rawTrimmedStart = raw.trimStart();
          if (
            /^`{1,2}(?:json|ts|typescript|javascript|js)?[\s\n]*$/i.test(
              rawTrimmedStart,
            )
          ) {
            return ""; // 等待更多 chunk 累積，讓 stripCodeFences 能正確識別
          }
 
          const trimmed = stripCodeFences(raw).trimStart();
          // 如果以 { 或 [ 開頭，這是純 JSON 回覆，不返回任何文字
          if (trimmed.startsWith("{") || trimmed.startsWith("[")) {
            return "";
          }
          // 非 JSON 開頭 — 可能是混合格式（前綴 Markdown + 尾部 JSON）
          // 找到 JSON 結構的起始位置（"response": 前的 { 或 [ ）
          const responseKeyIdx = trimmed.search(/"response"\s*:/);
          if (responseKeyIdx >= 0) {
            const jsonStart = trimmed.lastIndexOf("{", responseKeyIdx);
            // 只返回 JSON 前的文字部分
            return jsonStart > 0 ? trimmed.slice(0, jsonStart).trim() : "";
          }
          // 沒有 "response": 但可能有部分 JSON 結構正在形成
          // 安全檢查：如果文字末尾包含 { 後跟著疑似 JSON key 的片段，截斷之
          const possibleJsonStart = trimmed.search(/\{[\s]*"/);
          if (possibleJsonStart >= 0) {
            const beforeJson = trimmed.slice(0, possibleJsonStart).trim();
            return beforeJson || "";
          }
 
          // 🔥 Leakage Fix: 如果文字以 { 結尾，且可能是 JSON 的開頭
          // 我們必須緩衝這個字符，直到下一個 chunk 確認它不是 JSON
          // 改進：只在 { 前面是換行符、或是整段文字的開頭時才緩衝
          // 這樣可以避免誤殺 "math: f(x) = {" 這種句子中間的 {
          if (trimmed.endsWith("{")) {
            const lastBraceIndex = trimmed.lastIndexOf("{");
            // 如果 { 是第一個字符，或者 { 前面有換行符（忽略空白），則可能是回應結束後的 JSON
            const textBeforeBrace = trimmed.slice(0, lastBraceIndex).trimEnd();
 
            const isStartOfText = lastBraceIndex === 0;
            const isAfterNewline =
              raw.lastIndexOf("\n") < raw.lastIndexOf("{") &&
              textBeforeBrace.length < trimmed.length - 1; // 簡單檢查：raw 中 { 前面有換行
 
            // 更精確：檢查 trimmed 中 { 前面的內容是否為空（即 { 是整段開頭）
            // 或者 raw accumulation 中 { 是否位於新的一行
            const lastCharBeforeBrace = raw
              .slice(0, raw.lastIndexOf("{"))
              .trimEnd()
              .slice(-1);
 
            const indexInRaw = raw.lastIndexOf("{");
 
            if (indexInRaw === 0) {
              return ""; // 緩衝
            }
 
            const charBefore = raw[indexInRaw - 1];
            if (charBefore === "\n" || charBefore === "\r") {
              // 位於行首的 { -> 高機率是 JSON block 開始
              return trimmed.slice(0, lastBraceIndex).trimEnd();
            }
          }
 
          // 確定沒有 JSON 結構，返回整段文字
          return trimmed;
        }
 
        return result;
      };
 
      const processIncremental = () => {
        // 策略：永遠優先使用 extractRawTextContent() 進行即時串流
        // 這確保即使 bestEffortParse 失敗，使用者也能看到文字逐步出現
        // 只有當 JSON 結構完整解析成功時，才切換到結構化模式處理 template 等
        const rawText = extractRawTextContent();
 
        // 階段 1：嘗試 JSON 結構化解析（用於 template 等非文字 part）
        const content = getCleanContent();
        if (content) {
          const parts = parseResponseParts(content);
          if (Array.isArray(parts) && parts.length > 0) {
            if (!structuredModeActive) {
              structuredModeActive = true;
              // 切換到結構化模式：已送出的直接串流長度成為基準
              // 注意：如果之前因為緩衝而少送了 {，現在切換到結構化模式會由 template 處理，
              // 但如果是純文字回覆，我們需要確保不丟失字符。
              // 這裡的邏輯是 structuredMode 接管後，會從 parts 重新計算 sentTextLength
              sentTextLength = directStreamSentLength;
            }
 
            for (let i = sentPartsCount; i < parts.length; i++) {
              const part = parts[i];
              if (!part || !part.type) continue;
 
              if (part.type === "text") {
                const fullText = part.text ?? "";
                if (i === sentPartsCount && i === parts.length - 1) {
                  // 串流中最後一個文字 part — 發送增量
                  const delta = fullText.substring(sentTextLength);
                  if (delta.length > 0) {
                    enqueueSSE(new TextChunk(delta).toSSEString());
                    sentTextLength = fullText.length;
                    directStreamSentLength = rawText.length; // 同步直接串流追蹤
                  }
                } else if (i >= sentPartsCount) {
                  // 完整 part — 發送剩餘文字
                  const delta = fullText.substring(
                    i === sentPartsCount ? sentTextLength : 0,
                  );
                  if (delta.length > 0) {
                    enqueueSSE(new TextChunk(delta).toSSEString());
                  }
                  sentTextLength = 0;
                  sentPartsCount = i + 1;
                  directStreamSentLength = rawText.length;
                }
              } else if (part.type === "template" && part.name) {
                // 非最後一個 part 的 template：直接 emit（資料已完整）
                // 最後一個 part 的 template：只在 JSON 結構已完整閉合時才 emit
                //
                // 背景：原始設計使用 i < parts.length - 1 讓最後一個 template 只由
                // processFinal() 發射，但這會導致：
                // - 單一 DataTable template 回覆在整個 53 秒串流期間零 SSE 事件
                // - HMR recompile 中斷連線時 processFinal() 不執行 → 空流 → "本次回覆未成功生成內容"
                //
                // 改進：以「rawAccumulated 是否已完整閉合」取代「不是最後一個 part」的判斷：
                // - ]}  結尾：response 陣列 + 外層物件均已閉合 → JSON 完整 → 安全 emit
                // - 非最後一個 part：已有後續 part 證明此 template 資料完整 → 直接 emit
                // 若 JSON 尚未完整閉合（串流中途），跳過此次 emit，等待下次 chunk 觸發
                const isLastPart = i === parts.length - 1;
                const jsonComplete = /\]\s*\}\s*$/.test(
                  rawAccumulated.trimEnd(),
                );
 
                if (!isLastPart || jsonComplete) {
                  enqueueSSE(
                    new ResponseTemplate(
                      part.name,
                      part.templateProps ?? {},
                    ).toSSEString(),
                  );
                  sentPartsCount = i + 1;
                  sentTextLength = 0;
                }
                // else: JSON 尚未完整（串流中途），等待 processFinal() 確保完整資料後發射
              }
            }
            return; // 結構化模式處理完成
          }
        }
 
        // 階段 2：JSON 解析失敗或尚未就緒 — 直接串流原始文字
        // 這是關鍵路徑：確保 streaming 期間使用者能即時看到文字
        // ⚠️ 但若已偵測到 JSON 結構開始，即使解析尚未成功也要停止
        //   否則 JSON 原始碼（如 { "response": [）會洩漏到前端顯示
        if (!structuredModeActive && rawText.length > directStreamSentLength) {
          const stripped = stripCodeFences(rawAccumulated).trimStart();
          const jsonDetected = isJsonStructureDetected(
            rawAccumulated,
            stripped,
          );
 
          if (!jsonDetected) {
            const delta = rawText.substring(directStreamSentLength);
            enqueueSSE(new TextChunk(delta).toSSEString());
            directStreamSentLength = rawText.length;
          }
          // Do not send if jsonDetected=true: wait for structured mode to handle it after JSON is complete
        }
      };
 
      const processFinal = () => {
        const content = getCleanContent();
        if (!content) {
          // content 為空但 rawAccumulated 不為空 → 嘗試完整混合格式解析
          if (rawAccumulated.trim()) {
            const hybridParts = parseResponseParts(rawAccumulated);
            if (Array.isArray(hybridParts) && hybridParts.length > 0) {
              emitFinalParts(hybridParts);
            }
          }
          return;
        }
 
        const parts = parseResponseParts(content);
        if (!Array.isArray(parts) || parts.length === 0) {
          // 回退到完整原始內容的混合格式解析
          const hybridParts = parseResponseParts(rawAccumulated);
          if (Array.isArray(hybridParts) && hybridParts.length > 0) {
            emitFinalParts(hybridParts);
          }
          return;
        }
 
        emitFinalParts(parts);
      };
 
      const emitFinalParts = (parts: ResponsePart[]) => {
        for (let i = sentPartsCount; i < parts.length; i++) {
          const part = parts[i];
          if (!part || !part.type) continue;
 
          if (part.type === "text") {
            const fullText = part.text ?? "";
            // 計算已發送的文字長度（考慮直接串流模式已發送的部分）
            const alreadySent =
              i === sentPartsCount
                ? Math.max(
                  sentTextLength,
                  structuredModeActive ? 0 : directStreamSentLength,
                )
                : 0;
            const delta = fullText.substring(alreadySent);
            if (delta.length > 0) {
              enqueueSSE(new TextChunk(delta).toSSEString());
            }
          } else if (part.type === "template" && part.name) {
            enqueueSSE(
              new ResponseTemplate(
                part.name,
                part.templateProps ?? {},
              ).toSSEString(),
            );
          }
        }
        sentPartsCount = parts.length;
        sentTextLength = 0;
      };
 
      for await (const chunk of eventStream) {
        if (chunk.type === "tool-call") {
          if (chunk.toolName === "googleSearch" && !thinkingEmitted) {
            thinkingEmitted = true;
            enqueueSSE(
              new ResponseTemplate("thinking_process", {
                mode: "search",
              }).toSSEString(),
            );
          }
        }
 
        if (chunk.type === "text-delta") {
          rawAccumulated += chunk.text;
 
          if (!detectionDone) {
            const trimmed = rawAccumulated.trimStart();
            if (trimmed.startsWith("```")) {
              hasCodeBlock = true;
            }
            if (trimmed.length >= 4) {
              detectionDone = true;
            }
          }
 
          // P1: 節流 — 減少每 token 的 JSON 解析開銷
          charsSinceLastParse += chunk.text.length;
          const shouldFullParse =
            charsSinceLastParse >= PARSE_THROTTLE_CHARS ||
            chunk.text.includes("}") ||
            chunk.text.includes("]");
 
          if (shouldFullParse) {
            charsSinceLastParse = 0;
            processIncremental();
          } else if (!structuredModeActive) {
            // 直接串流模式下仍然即時發送文字（不需要 JSON 解析）
            const rawText = extractRawTextContent();
            if (rawText.length > directStreamSentLength) {
              const stripped = stripCodeFences(rawAccumulated).trimStart();
              const jsonDetected = isJsonStructureDetected(
                rawAccumulated,
                stripped,
              );
 
              if (!jsonDetected) {
                const delta = rawText.substring(directStreamSentLength);
                enqueueSSE(new TextChunk(delta).toSSEString());
                directStreamSentLength = rawText.length;
              }
            }
          }
        }
      }
 
      processFinal();
 
      state.controller?.close();
    } catch (error) {
      if (error instanceof Error) {
        state.controller?.error(error);
      } else {
        state.controller?.error(new Error(String(error)));
      }
    }
  })();
 
  return readableStream;
}
 
/**
 * Convert NexusMind messages format to Crayon-compatible messages format.
 * Crayon expects { role, message } while NexusMind uses { role, content }.
 */
export function toCrayonMessages(
  messages: Array<{ role: string; content: string }>,
) {
  return messages.map((m) => {
    if (m.role === "user") {
      return {
        role: "user" as const,
        message: m.content,
      };
    }
    // For assistant messages, try to parse as Crayon format first
    const parsedParts = parseResponseParts(m.content);
    if (parsedParts) {
      return {
        role: "assistant" as const,
        message: parsedParts,
      };
    }
    return {
      role: "assistant" as const,
      message: [{ type: "text" as const, text: m.content }],
    };
  });
}
