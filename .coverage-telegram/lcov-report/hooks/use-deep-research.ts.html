"use client";
 
import { useState, useCallback, useRef, useEffect } from "react";
import type { ResearchEngine } from "@/types/research";
import { ENGINE_METADATA, ENGINE_API_ROUTES } from "@/types/research";
 
const REPORT_STORAGE_KEY = "nexusmind_deep_research_report";
 
export function loadPersistedReport(): ResearchReport | null {
  if (typeof window === "undefined") return null;
  try {
    const raw = localStorage.getItem(REPORT_STORAGE_KEY);
    if (!raw) return null;
    return JSON.parse(raw) as ResearchReport;
  } catch {
    return null;
  }
}
 
export function persistReport(report: ResearchReport | null) {
  if (typeof window === "undefined") return;
  try {
    if (report) {
      localStorage.setItem(REPORT_STORAGE_KEY, JSON.stringify(report));
    } else {
      localStorage.removeItem(REPORT_STORAGE_KEY);
    }
  } catch {
    // localStorage 已滿或不可用時靜默失敗
  }
}
 
interface SubQuestion {
  id: string;
  question: string;
  searchQuery: string;
  priority: string;
  status?: "decomposing" | "pending" | "searching" | "completed";
}
 
interface ReportSection {
  heading: string;
  content: string;
}
 
export interface ResearchReport {
  title: string;
  executiveSummary: string;
  sections: ReportSection[];
  conclusion: string;
  sources: { title: string; url: string }[];
  generatedAt: string;
}
 
interface ProgressEvent {
  stage: string;
  message: string;
  current?: number;
  total?: number;
  subQuestions?: SubQuestion[];
  mainTopic?: string;
}
 
interface UseDeepResearchReturn {
  topic: string;
  setTopic: (topic: string) => void;
  engine: ResearchEngine;
  setEngine: (engine: ResearchEngine) => void;
  loading: boolean;
  progress: ProgressEvent | null;
  subQuestions: SubQuestion[];
  report: ResearchReport | null;
  error: string | null;
  startResearch: (overrideTopic?: string, context?: string) => Promise<void>;
  abort: () => void;
  reset: () => void;
  showSubQuestions: boolean;
  setShowSubQuestions: (show: boolean) => void;
}
 
export function useDeepResearch(
  conversationId?: string,
): UseDeepResearchReturn {
  const [topic, setTopic] = useState("");
  const [engine, setEngine] = useState<ResearchEngine>("gemini");
  const [loading, setLoading] = useState(false);
  const [progress, setProgress] = useState<ProgressEvent | null>(null);
  const [subQuestions, setSubQuestions] = useState<SubQuestion[]>([]);
  const [report, setReportState] = useState<ResearchReport | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [showSubQuestions, setShowSubQuestions] = useState(false);
  const abortRef = useRef<AbortController | null>(null);
 
  // 封裝 setReport，同時持久化到 localStorage
  const setReport = useCallback((r: ResearchReport | null) => {
    setReportState(r);
    persistReport(r);
  }, []);
 
  // 初始化時從 localStorage 還原上次的報告
  useEffect(() => {
    const persisted = loadPersistedReport();
    if (persisted) {
      setReportState(persisted);
      setProgress({ stage: "completed", message: "研究完成！" });
    }
  }, []);
 
  const handleSSEEvent = useCallback(
    (event: string, data: Record<string, unknown>) => {
      if (event === "progress") {
        const progressData = data as unknown as ProgressEvent;
        setProgress(progressData);
 
        // 處理拆解階段的 partial streaming（子問題逐個出現）
        const partialSubs = data.partialSubQuestions as
          | Partial<SubQuestion>[]
          | undefined;
        if (
          progressData.stage === "decomposing" &&
          partialSubs &&
          partialSubs.length > 0
        ) {
          setSubQuestions(
            partialSubs
              .filter(
                (sq): sq is SubQuestion =>
                  !!sq.id && !!sq.question && !!sq.searchQuery,
              )
              .map((sq) => ({
                ...sq,
                status: "decomposing" as const,
              })),
          );
        }
 
        if (progressData.subQuestions) {
          // 拆解完成後，用完整的子問題列表覆蓋，所有狀態設為 pending
          setSubQuestions(
            progressData.subQuestions.map((sq) => ({
              ...sq,
              status: "pending" as const,
            })),
          );
        }
        // 更新各子問題的搜尋狀態
        const currentId = data.currentQuestionId as string | undefined;
        const completedIds = data.completedQuestionIds as string[] | undefined;
        if (currentId || (completedIds && completedIds.length > 0)) {
          setSubQuestions((prev) =>
            prev.map((sq) => {
              if (completedIds?.includes(sq.id)) {
                return { ...sq, status: "completed" as const };
              }
              if (sq.id === currentId) {
                return { ...sq, status: "searching" as const };
              }
              // 保持已完成的狀態，其餘為 pending
              if (sq.status === "completed") return sq;
              return { ...sq, status: "pending" as const };
            }),
          );
        }
      } else if (event === "complete") {
        setReport(data.report as ResearchReport);
        setProgress({ stage: "completed", message: "研究完成！" });
        // 所有子問題標記完成
        setSubQuestions((prev) =>
          prev.map((sq) => ({ ...sq, status: "completed" as const })),
        );
      } else if (event === "error") {
        setError(data.message as string);
      }
    },
    [setReport],
  );
 
  const startResearch = useCallback(
    async (overrideTopic?: string, context?: string) => {
      const researchTopic = overrideTopic ?? topic;
      if (!researchTopic.trim() || loading) return;
 
      // Abort previous request if any
      if (abortRef.current) {
        abortRef.current.abort();
      }
      const controller = new AbortController();
      abortRef.current = controller;
 
      setLoading(true);
      setError(null);
      setReport(null);
      setSubQuestions([]);
 
      const firstStage = ENGINE_METADATA[engine].stages[0];
      setProgress({
        stage: firstStage,
        message: `正在連線${ENGINE_METADATA[engine].name}引擎...`,
      });
 
      try {
        const body: Record<string, unknown> = {
          topic: researchTopic,
          conversationId,
        };
        if (context) {
          body.context = context;
        }
 
        const response = await fetch(ENGINE_API_ROUTES[engine], {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(body),
          signal: controller.signal,
        });
 
        if (!response.ok) {
          const data = await response.json();
          throw new Error(data.error || "啟動研究失敗");
        }
 
        const reader = response.body?.getReader();
        if (!reader) throw new Error("無法建立串流連線");
 
        const decoder = new TextDecoder();
        let buffer = "";
 
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
 
          buffer += decoder.decode(value, { stream: true });
          const lines = buffer.split("\n");
          buffer = lines.pop() ?? "";
 
          let currentEvent = "";
          for (const line of lines) {
            if (line.startsWith("event:")) {
              currentEvent = line.slice(6).trim();
            } else if (line.startsWith("data:") && currentEvent) {
              try {
                const data = JSON.parse(line.slice(5));
                handleSSEEvent(currentEvent, data);
              } catch {
                // skip malformed JSON
              }
              currentEvent = "";
            }
          }
        }
      } catch (err) {
        if (err instanceof DOMException && err.name === "AbortError") return;
        setError(err instanceof Error ? err.message : "研究過程發生錯誤");
      } finally {
        setLoading(false);
        abortRef.current = null;
      }
    },
    [topic, engine, loading, conversationId, handleSSEEvent, setReport],
  );
 
  const abort = useCallback(() => {
    if (abortRef.current) {
      abortRef.current.abort();
      abortRef.current = null;
    }
    setLoading(false);
    setProgress({ stage: "aborted", message: "研究已中止" });
  }, []);
 
  const reset = useCallback(() => {
    if (abortRef.current) {
      abortRef.current.abort();
      abortRef.current = null;
    }
    setTopic("");
    setLoading(false);
    setProgress(null);
    setSubQuestions([]);
    setReport(null);
    setError(null);
  }, [setReport]);
 
  return {
    topic,
    setTopic,
    engine,
    setEngine,
    loading,
    progress,
    subQuestions,
    report,
    error,
    startResearch,
    abort,
    reset,
    showSubQuestions,
    setShowSubQuestions,
  };
}
 
export function formatReportAsMarkdown(report: ResearchReport): string {
  const parts = [
    `# ${report.title}`,
    "",
    "## 執行摘要",
    "",
    report.executiveSummary,
    "",
    ...report.sections.flatMap((s) => [`## ${s.heading}`, "", s.content, ""]),
    "## 結論與建議",
    "",
    report.conclusion,
    "",
    "## 參考來源",
    "",
    ...report.sources.map((s) => `- [${s.title}](${s.url})`),
  ];
  return parts.join("\n");
}
